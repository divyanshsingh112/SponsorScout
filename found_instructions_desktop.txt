const fs = require('fs');
const path = require('path');

const desktopDir = 'c:\\Users\\sdivy\\OneDrive\\Desktop';

function walkDir(dir, callback) {
  try {
    const files = fs.readdirSync(dir);
    for (const file of files) {
      const filePath = path.join(dir, file);
      let stat;
      try {
        stat = fs.statSync(filePath);
      } catch (e) {
        continue;
      }
      if (stat.isDirectory()) {
        walkDir(filePath, callback);
      } else {
        callback(filePath);
      }
    }
  } catch (e) {
    // Ignore error
  }
}

console.log('Searching all files in Desktop recursively...');
let foundCount = 0;

walkDir(desktopDir, (filePath) => {
  if (filePath.endsWith('.md') || filePath.endsWith('.txt') || filePath.endsWith('.json') || filePath.endsWith('.js') || filePath.endsWith('.ts')) {
    try {
      const content = fs.readFileSync(filePath);
      const str = content.toString('utf8');
      if (str.includes('ANTIGRAVITY INSTRUCTION') && str.includes('v4.0')) {
        console.log(`>>> FOUND MATCHING KEYWORDS IN FILE: ${filePath}`);
        foundCount++;
        
        // Save to file
        fs.writeFileSync('found_instructions_desktop.txt', str, 'utf8');
        console.log('Saved to found_instructions_desktop.txt!');
      }
    } catch (e) {
      // Ignore read errors
    }
  }
});

console.log(`Done! Found ${foundCount} matching files.`);
